130  public async Task<Entity> UploadFiles()
131  {
132  processedFilesCounter = 0;
133  try
134  {
135  // perform handshake if needed
136  Batch = Batch ?? client.Batch().Result;
137  }
138  catch (ServerException exception)
139  {
140  throw new FailedHandshakeException("Failed to initialize batch with the server.", exception);
141  }
142 
143  lock (syncFiles)
144  {
145  Parallel.ForEach(filesToUpload, new ParallelOptions { MaxDegreeOfParallelism = NumConcurrentUploads }, path =>
146  {
147  if (!File.Exists(path))
148  {
149  throw new FileNotFoundException(path);
150  }
151 
152  int i;
153  lock (syncCounter)
154  {
155  i = processedFilesCounter++;
156  }
157 
158  if (IsChunkedUpload)
159  {
160  FileStream fs = File.OpenRead(path);
161  int readBytes;
162  byte[] buffer = new byte[ChunkSize];
163  UploadJob job;
164  List<Blob> blobs = new List<Blob>();
165 
166  while ((readBytes = fs.Read(buffer, 0, buffer.Length)) > 0)
167  {
168  byte[] temp = new byte[readBytes];
169  Array.Copy(buffer, temp, readBytes);
170  blobs.Add(new Blob(Path.GetFileName(path)).SetContent(temp));
171  Array.Clear(buffer, 0, ChunkSize);
172  }
173 
174  int chunksProcessed = 0;
175  foreach (Blob blob in blobs)
176  {
177  job = new UploadJob(blob).SetFileId(i)
178  .SetChunkIndex(chunksProcessed)
179  .SetChunkCount(blobs.Count);
180  UploadBlob(job, true);
181  chunksProcessed++;
182  }
183  }
184  else
185  {
186  byte[] contents = File.ReadAllBytes(path);
187  Blob blob = new Blob(Path.GetFileName(path)).SetContent(contents);
188  UploadJob job = new UploadJob(blob).SetFileId(i);
189  UploadBlob(job);
190  }
191  });
192  }
193  return await Batch.Info();
194  }

197  {

207  }
208  }
209  catch (ServerException exception)

211  throw new FailedToUploadException(job.ToString(), exception);
