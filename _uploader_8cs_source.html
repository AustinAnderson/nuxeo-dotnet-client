164  private async Task<Batch> ProcessFile(string path)
165  {
166  Batch batch = null;
167  await semaphore.WaitAsync();
168  try
169  {
170  if (!File.Exists(path))
171  {
172  throw new FileNotFoundException(path);
173  }
174 
175  int i;
176  lock (syncCounter)
177  {
178  i = processedFilesCounter++;
179  }
180 
181  if (IsChunkedUpload)
182  {
183  FileStream fs = File.OpenRead(path);
184  int readBytes;
185  byte[] buffer = new byte[ChunkSize];
186  UploadJob job;
187  List<Blob> blobs = new List<Blob>();
188 
189  while ((readBytes = fs.Read(buffer, 0, buffer.Length)) > 0)
190  {
191  byte[] temp = new byte[readBytes];
192  Array.Copy(buffer, temp, readBytes);
193  blobs.Add(new Blob(Path.GetFileName(path)).SetContent(temp));
194  Array.Clear(buffer, 0, ChunkSize);
195  }
196 
197  int chunksProcessed = 0;
198  foreach (Blob blob in blobs)
199  {
200  job = new UploadJob(blob).SetFileId(i)
201  .SetChunkIndex(chunksProcessed)
202  .SetChunkCount(blobs.Count);
203  batch = await UploadBlob(job, true);
204  chunksProcessed++;
205  }
206  }
207  else
208  {
209  byte[] contents = File.ReadAllBytes(path);
210  Blob blob = new Blob(Path.GetFileName(path)).SetContent(contents);
211  UploadJob job = new UploadJob(blob).SetFileId(i);
212  batch = await UploadBlob(job);
213  }
214  }
215  finally
216  {
217  semaphore.Release();
218  }
219  return batch;
220  }
