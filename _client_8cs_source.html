498  MultipartContent requestContent = new MultipartContent("related");
499 
500  string firstPartStr = JsonConvert.SerializeObject(data);
501  HttpContent firstPart = new StringContent(firstPartStr);
502  firstPart.Headers.ContentType = new MediaTypeHeaderValue(Client.ContentType.NXREQUEST);
503  firstPart.Headers.Add("Content-Transfer-Encoding", "8bit");
504  firstPart.Headers.Add("Content-ID", "request");
505  requestContent.Add(new StringContent(firstPartStr));
506  JArray inputObjs = (input is JArray ? input.ToObject<JArray>() : new JArray { input });
507  foreach (JToken obj in inputObjs)
508  {
509  if (obj["filename"] == null ||
510  obj["content"] == null ||
511  obj["mimetype"] == null)
512  {
513  throw new HttpRequestException("Invalid blob file.");
514  }
515  HttpContent part = new ByteArrayContent(obj["content"].ToObject<byte[]>());
516  part.Headers.ContentType = new MediaTypeHeaderValue(obj["mimetype"].ToObject<string>());
517  part.Headers.ContentDisposition = new ContentDispositionHeaderValue("attachment")
518  {

520  };
521  part.Headers.Add("Content-Transfer-Encoding", "binary");
522  part.Headers.Add("Content-ID", "input");
523  requestContent.Add(part);
524  }
525  request.Content = requestContent;
526 
527  return await ProcessRequest(request, additionalHeaders);

548  private async Task<JToken> ProcessResponse(HttpResponseMessage response)
549  {
550  JToken json = null;
551  await response.Content.ReadAsStringAsync().ContinueWith(output =>
552  {
553  string result = output.Result;
554  if (result != string.Empty)
555  {
556  try
557  {
558  json = JToken.Parse(result);
559  }
560  catch (JsonReaderException)
561  {
562  json = response.IsSuccessStatusCode ? null : new JObject { { "message", result } };
563  }
564  if ((int)response.StatusCode >= 400 && (int)response.StatusCode <= 599)
565  {
566  if (response.StatusCode == System.Net.HttpStatusCode.NotFound &&
567  json["code"] != null &&
568  json["code"].ToObject<string>().Contains("DocumentNotFoundException"))
569  {
570  json = null;
571  }
572  else
573  {
574  if (json["message"] != null)
575  {
576  throw new ServerException(response.StatusCode, json["message"].ToObject<string>());
577  }
578  else
579  {
580  throw new ServerException(response.StatusCode, json.ToObject<string>());
581  }
582  }
583  }
584  }
585  });
586  return json;
