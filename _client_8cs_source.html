389  internal async Task<Entity> RequestMultipart(string endpoint,
390  JToken data,
391  JToken input,
392  HttpMethod htttpMethod = null,
393  Dictionary<string, string> additionalHeaders = null,
394  string contentType = ContentType.NXREQUEST)
395  {

400  if (input == null)

402  throw new NullReferenceException("input parameter was null.");
403  }

405  HttpRequestMessage request = new HttpRequestMessage(htttpMethod ?? HttpMethod.Post, endpoint);
406  MultipartContent requestContent = new MultipartContent("related");
407 
408  string firstPartStr = JsonConvert.SerializeObject(data);
409  HttpContent firstPart = new StringContent(firstPartStr);
410  firstPart.Headers.ContentType = new MediaTypeHeaderValue(Client.ContentType.NXREQUEST);
411  firstPart.Headers.Add("Content-Transfer-Encoding", "8bit");
412  firstPart.Headers.Add("Content-ID", "request");
413  requestContent.Add(new StringContent(firstPartStr));
414  JArray inputObjs = (input is JArray ? input.ToObject<JArray>() : new JArray { input });
415  foreach (JToken obj in inputObjs)
416  {
417  if (obj["filename"] == null ||
418  obj["content"] == null ||
419  obj["mimetype"] == null)
420  {
421  throw new HttpRequestException("Invalid blob file.");
422  }
423  HttpContent part = new ByteArrayContent(obj["content"].ToObject<byte[]>());
424  part.Headers.ContentType = new MediaTypeHeaderValue(obj["mimetype"].ToObject<string>());
425  part.Headers.ContentDisposition = new ContentDispositionHeaderValue("attachment")
426  {
427  FileName = obj["filename"].ToObject<string>()
428  };
429  part.Headers.Add("Content-Transfer-Encoding", "binary");
430  part.Headers.Add("Content-ID", "input");
431  requestContent.Add(part);
432  }
433  request.Content = requestContent;
434

438  internal async Task<Entity> ProcessRequest(HttpRequestMessage request, Dictionary<string, string> additionalHeaders = null)

456  private async Task<JToken> ProcessResponse(HttpResponseMessage response)
457  {
458  JToken json = null;
459  await response.Content.ReadAsStringAsync().ContinueWith(output =>
460  {
461  string result = output.Result;
462  if (result != string.Empty)
463  {
464  try
465  {
466  json = JToken.Parse(result);
467  }
468  catch (JsonReaderException)
469  {
470  json = response.IsSuccessStatusCode ? null : new JObject { { "message", result } };
471  }
472  if ((int)response.StatusCode >= 400 && (int)response.StatusCode <= 599)
473  {
474  if (response.StatusCode == System.Net.HttpStatusCode.NotFound &&
475  json["code"] != null &&
476  json["code"].ToObject<string>().Contains("DocumentNotFoundException"))
477  {
478  json = null;
479  }
480  else
481  {
482  if (json["message"] != null)
483  {
484  throw new ServerException(response.StatusCode, json["message"].ToObject<string>());
485  }
486  else
487  {
488  throw new ServerException(response.StatusCode, json.ToObject<string>());
489  }
490  }
491  }
492  }
493  });
494  return json;
495  }
