510  protected Dictionary<string, string> GenerateHeaders()
511  {
512  Dictionary<string, string> headers;
513  if (AdditionalHeaders != null)
514  {
515  headers = new Dictionary<string, string>(AdditionalHeaders);
516  }
517  else
518  {
519  headers = new Dictionary<string, string>();
520  }
521  if (Schemas != null && Schemas.Count > 0)
522  {
523  headers["X-NXDocumentProperties"] = string.Join(",", Schemas);
524  }
525  if (ContentEnrichers != null && ContentEnrichers.Count > 0)
526  {
527  headers["X-NXenrichers.document"] = string.Join(",", ContentEnrichers);
528  }
529  if (Repository != null && Repository != string.Empty)
530  {
531  headers["X-NXRepository"] = Repository;
532  }

536  private string GenerateEndpoint(bool withAdapter = false)
537  {
538  bool noUid = string.IsNullOrEmpty(Uid);
539  bool noPath = string.IsNullOrEmpty(Path);
540  if (noUid && noPath)
541  {
542  throw new InvalidEntityException("No remote document identifier was especified.");
543  }
544  else
545  {
546  return BuildEndpointForDoc((noPath ? Uid : Path), (noPath ? EndpointType.UID : EndpointType.PATH), withAdapter);
547  }

550  private string BuildEndpointForDoc(string doc, EndpointType type = EndpointType.UID, bool withAdapter = false)
551  {
552  string endpoint = client.ServerURL;
553  endpoint += client.RestPath;
554  endpoint += ((Repository != null && Repository != string.Empty) ? "repo/" + Repository + "/" : string.Empty);
555  endpoint += (type == EndpointType.UID ? "id" : "path");
556  endpoint += (doc.Length > 0 && doc[0] != '/' ? "/" : string.Empty) + doc;
557  if (withAdapter)
558  {
559  endpoint += (Adapter != null ? "/" + Adapter.GetEndpointSuffix() : string.Empty);
560  }
561  return endpoint;
562  }
