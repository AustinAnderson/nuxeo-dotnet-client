575  protected Dictionary<string, string> GenerateHeaders()
576  {
577  Dictionary<string, string> headers;
578  if (AdditionalHeaders != null)
579  {
580  headers = new Dictionary<string, string>(AdditionalHeaders);
581  }
582  else
583  {
584  headers = new Dictionary<string, string>();
585  }
586  if (Schemas != null && Schemas.Count > 0)
587  {
588  headers["X-NXDocumentProperties"] = string.Join(",", Schemas);
589  }
590  if (ContentEnrichers != null && ContentEnrichers.Count > 0)
591  {
592  headers["X-NXenrichers.document"] = string.Join(",", ContentEnrichers);
593  }
594  if (Repository != null && Repository != string.Empty)
595  {
596  headers["X-NXRepository"] = Repository;
597  }
598  return headers;

601  private string GenerateEndpoint(bool withAdapter = false, string extension = "")
602  {
603  bool noUid = string.IsNullOrEmpty(Uid);
604  bool noPath = string.IsNullOrEmpty(Path);
605  if (noUid && noPath)
606  {
607  throw new InvalidEntityException("No remote document identifier was especified.");
608  }
609  else
610  {
611  return BuildEndpointForDoc((noPath ? Uid : Path), extension, (noPath ? EndpointType.UID : EndpointType.PATH), withAdapter);
612  }
613  }

615  private string BuildEndpointForDoc(string doc, string extension = "", EndpointType type = EndpointType.UID, bool withAdapter = false)
616  {
617  List<string> segments = new List<string>();
618 
619  segments.Add(client.RestPath);
620  if (!string.IsNullOrEmpty(Repository))
621  {
622  segments.Add("repo");
623  segments.Add(Repository);
624  }
625  segments.Add(type == EndpointType.UID ? "id" : "path");
626  segments.Add(doc);
627  if (withAdapter && Adapter != null)
628  {
629  segments.Add(Adapter.GetEndpointSuffix());
630  }
631  if (!string.IsNullOrEmpty(extension))
632  {
633  segments.Add(extension);
634  }
635 
636  return UrlCombiner.Combine(segments.ToArray());
637  }
