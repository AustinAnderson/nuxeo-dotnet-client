106  public JObject Marshal(Entity entity)
107  {
108  JObject json = JObject.Parse(JsonConvert.SerializeObject(entity, new JsonSerializerSettings
109  {
110  NullValueHandling = NullValueHandling.Ignore,
111  DefaultValueHandling = DefaultValueHandling.Ignore
112  }));
113  if (businessObjects.ContainsKey(entity.EntityType))
114  {
115  JObject value = new JObject();
116  List<string> toRemove = new List<string>();
117  foreach (KeyValuePair<string, JToken> pair in json)
118  {
119  if (pair.Key == "entity-type")
120  continue;
121  value.Add(pair.Key, pair.Value);
122  toRemove.Add(pair.Key);
123  }
124  foreach (string property in toRemove)
125  {
126  json.Remove(property);
127  }
128  json.Add("value", value);
129  }
130  return json;
131  }

150  private Entity UnMarshalJObject(JObject jObj)
151  {
152  Entity result = null;
153  if (jObj["entity-type"] != null)
154  {
155  string entityType = jObj["entity-type"].ToObject<string>();
156  if (entityType == "documents")
157  {
158  if (jObj["isPaginable"] != null &&
159  jObj["isPaginable"].ToObject<bool>() == true)
160  {
161  Pageable page = JsonConvert.DeserializeObject<Pageable>(jObj.ToString());
162  foreach (Document document in page.Entries)
163  {
164  document.SetClient(client);
165  }
166  result = page;
167  }
168  else
169  {
170  Documents docs = JsonConvert.DeserializeObject<Documents>(jObj.ToString());
171  foreach (Document document in docs.Entries)
172  {
173  document.SetClient(client);
174  }
175  result = docs;
176  }
177  }
178  else if (entityType == "document")
179  {
180  result = JsonConvert.DeserializeObject<Document>(jObj.ToString()).SetClient(client);
181  }
182  else if (entityType == "null")
183  {
184  result = null;
185  }
186  else if (entityMap.ContainsKey(entityType))
187  {
188  result = (Entity)JsonConvert.DeserializeObject(jObj.ToString(), entityMap[entityType]);
189  }
190  else if (businessObjects.ContainsKey(entityType))
191  {
192  if (jObj["value"] != null && jObj["value"] is JObject)
193  {
194  JObject boJobj = new JObject();
195  boJobj.Add("entity-type", entityType);
196  foreach (KeyValuePair<string, JToken> pair in jObj["value"].ToObject<JObject>())
197  {
198  boJobj.Add(pair.Key, pair.Value);
199  }
200  result = (Entity)JsonConvert.DeserializeObject(boJobj.ToString(), businessObjects[entityType]);
201  }
202  else
203  {
204  throw new InvalidEntityException(jObj.ToString());
205  }
206  }
207  else
208  {
209  result = new UnknowEntity(jObj);
210  result.EntityType = entityType;
211  }
212  }
213  else if (jObj["batchId"] != null)
214  {
215  result = JsonConvert.DeserializeObject<Batch>(jObj.ToString()).SetClient(client).SetClient(client);
216  }
217  else if (jObj["name"] != null &&
218  jObj["size"] != null &&
219  jObj["uploadType"] != null)
220  {
221  result = JsonConvert.DeserializeObject<BatchFile>(jObj.ToString());
222  }
223  else
224  {
225  result = new UnknowEntity(jObj);
226  }
227  return result;
228  }
229  }
