347  public async Task<Entity> Execute()
348  {
349  Entity result = null;
350 
351  Dictionary<string, string> headers;
352  if (AdditionalHeaders != null)
353  {
354  headers = new Dictionary<string, string>(AdditionalHeaders);
355  }
356  else
357  {
358  headers = new Dictionary<string, string>();
359  }
360  if (Schemas != null && Schemas.Count > 0)
361  {
362  headers["X-NXDocumentProperties"] = string.Join(",", Schemas);
363  }
364  headers["Nuxeo-Transaction-Timeout"] = Timeout.ToString();
365  if (Repository != null && Repository != string.Empty)
366  {
367  headers["X-NXRepository"] = Repository;
368  }
369 
370  JObject data = new JObject();
371  if (Parameters != null)
372  {
373  JObject jsonParams = new JObject();
374  foreach (KeyValuePair<string, JToken> param in Parameters)
375  {
376  jsonParams.Add(param.Key, param.Value);
377  }
378  data.Add("params", jsonParams);
379  }
380  if (Context != null)
381  {
382  JObject jsonCotext = new JObject();
383  foreach (KeyValuePair<string, JToken> param in Context)
384  {
385  jsonCotext.Add(param.Key, param.Value);
386  }
387  data.Add("context", jsonCotext);
388  }
389  if (Input != null && !IsMultipart)
390  {
391  data.Add("input", Input);
392  }
393 
394  if (IsMultipart)
395  {
396  await client.RequestMultipart(Endpoint, data, Input, HttpMethod.Post, headers).ContinueWith(output =>
397  {
398  result = output.Result;
399  });
400  }
401  else
402  {
403  await client.RequestJson(Endpoint, data, HttpMethod.Post, headers).ContinueWith(output =>
404  {
405  result = output.Result;
406  });
407  }
408 
409  if (result is Document)
410  {
411  result = ((Document)result).SetClient(client);
412  }
413 
414  return result;
415  }
416  }
