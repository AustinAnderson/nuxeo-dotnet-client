348  public async Task<Entity> Execute()
349  {
350  Entity result = null;
351 
352  Dictionary<string, string> headers;
353  if (AdditionalHeaders != null)
354  {
355  headers = new Dictionary<string, string>(AdditionalHeaders);
356  }
357  else
358  {
359  headers = new Dictionary<string, string>();
360  }
361  if (Schemas != null && Schemas.Count > 0)
362  {
363  headers["X-NXDocumentProperties"] = string.Join(",", Schemas);
364  }
365  headers["Nuxeo-Transaction-Timeout"] = Timeout.ToString();
366  if (Repository != null && Repository != string.Empty)
367  {
368  headers["X-NXRepository"] = Repository;
369  }
370 
371  JObject data = new JObject();
372  if (Parameters != null)
373  {
374  JObject jsonParams = new JObject();
375  foreach (KeyValuePair<string, JToken> param in Parameters)
376  {
377  jsonParams.Add(param.Key, param.Value);
378  }
379  data.Add("params", jsonParams);
380  }
381  if (Context != null)
382  {
383  JObject jsonCotext = new JObject();
384  foreach (KeyValuePair<string, JToken> param in Context)
385  {
386  jsonCotext.Add(param.Key, param.Value);
387  }
388  data.Add("context", jsonCotext);
389  }
390  if (Input != null && !IsMultipart)
391  {
392  data.Add("input", Input);
393  }
394 
395  if (IsMultipart)
396  {
397  await client.RequestMultipart(Endpoint, data, Input, HttpMethod.Post, headers).ContinueWith(output =>
398  {
399  result = output.Result;
400  });
401  }
402  else
403  {
404  await client.RequestJson(Endpoint, null, data, HttpMethod.Post, headers).ContinueWith(output =>
405  {
406  result = output.Result;
407  });
408  }
409 
410  return result;
411  }
412  }
