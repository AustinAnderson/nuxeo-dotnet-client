130  {

132  int parts = (int)Math.Ceiling(Content.Length / (double)numChunks);
133  for (int i = 0; i < numChunks; i++)
134  {
135  result[i] = new Blob(Filename,
136  Content.SubArray<byte>(i * parts, (int)Math.Min(parts, Content.Length - (i * parts))),
137  MimeType);

139  }
